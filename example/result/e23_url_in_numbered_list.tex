\chapter{URL in numbered point list example}
\begin{enumerate}
\item A numbered point with a URL: \penalty0\href{https://bitbucket.org/tom-bjorkholm/mformat/src/master/example/src}{This URL link to the examples.}\penalty0
\item A numbered point with a bold URL: \textbf{\penalty0\href{https://bitbucket.org/tom-bjorkholm/mformat/src/master/example/src}{This bold URL link to the example source code.}\penalty0}
\item And with an italic URL: \textit{\penalty0\href{https://bitbucket.org/tom-bjorkholm/mformat/src/master/example/result}{This italic URL link to the examples result.}\penalty0}
\item And with an italic and bold URL: \textit{\textbf{\penalty0\href{https://bitbucket.org/tom-bjorkholm/mformat/src/master/example/src}{This italic and bold URL link to the examples.}\penalty0}}
\end{enumerate}
